    let answer = 0;
    let q1 = queue1.reduce((acc,curr) => acc+curr);
    let q2 =  queue2.reduce((acc,curr) => acc+curr);

        this.headIndex = 0;
        this.tailIndex = 0;

        this.items[this.tailIndex] = item;
        this.tailIndex++;

        delete this.items[this.headIndex];
