function gcd(a, b) {
  const remainder = a % b;

function solution(arr) {
  const gcd = (a, b) => (a % b ? gcd(b, a % b) : b);
  let answer = arr[0];
  for (let i = 1; i < arr.length; i++) {
    answer = (arr[i] * answer) / gcd(arr[i], answer);
  }
  return answer;
}
