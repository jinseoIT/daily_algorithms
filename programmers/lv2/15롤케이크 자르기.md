function solution(topping) {
    let answer = 0;
    const personA = new Map();
    const personB = new Map();
    
    topping.forEach(v => {
        if(personA.has(v)){
            personA.set(v, personA.get(v)+1);
            return;
        } 
        personA.set(v, 1);
    })

            personA.set(v, personA.get(v) -1);
        } else {
            personA.delete(v);
        }
        personB.set(v,1);
        if(personA.size == personB.size){
            answer++;
        }
    })
    return answer;
}
