function solution(priorities, location) {
  const arr = priorities.map((v, i) => {
    return [v, i];
  });
  let cnt = 0;
  while (arr.length) {
    const maxNum = Math.max(...arr.map((v) => v[0]));
    const [num, loc] = arr.shift();
    if (maxNum <= num) {
      cnt++;
      if (loc == location) {
        return cnt;
      }
    } else {
      arr.push([num, loc]);
    }
  }
}
