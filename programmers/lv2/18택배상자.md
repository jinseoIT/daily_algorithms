function solution(order) {
    let answer = 0;
    let box = 1;
    const Len = order.length+1
    const container = [];
    order.reverse();
    
    while(order.length > 0){
        const lastNum = container.at(-1);
        if(order[order.length-1] == box){
            answer++;
            box++;
            order.pop();
        }else if(order[order.length-1] == lastNum){
            answer++;
            container.pop();
            order.pop();
        } else {
            container.push(box);
            box++;
        }
        if(box > Len+1) break;
    }
    return answer;
}

function solution(order) {
  let idx = 0;

    }
  }
