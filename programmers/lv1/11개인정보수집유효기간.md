function solution(today, terms, privacies) {
    const nowDt = new Date(today);
    const termObj = {}
    for(const term of terms){
        const [type, month] = term.split(' ');
        termObj[type] = +month;
    }
    const isExpired = (privacy) => {
        const [date, type] = privacy.split(' ');
        const [yyyy,mm,dd] = date.split('.').map(Number);

        let year = yyyy + Math.floor(termObj[type] / 12), 
            month = mm + termObj[type] % 12, 
            day = dd - 1;

        if (month > 12) {
            year += 1;
            month %= 12;
        } 

        if (day === 0) {
            day = 28;
            month -= 1;
        }

        return nowDt > new Date(`${year}.${month}.${day}`);
    }
    const answer = [];
    privacies.forEach((privacy, idx) => {
        if(isExpired(privacy)) answer.push(idx+1);
    })
    return answer;
}
