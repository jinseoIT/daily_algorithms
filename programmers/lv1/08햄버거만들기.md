function solution(ingredient) {
    let answer = 0;
    let index = 0;
    
    while(true){
        if(ingredient[index] == 1 && ingredient[index+1] == 2 && ingredient[index+2] == 3 && ingredient[index+3] == 1){
            answer++;
            ingredient.splice(index, 4)
            index = 0;
        } else {
            if(index > ingredient.length-4) break;
            if(ingredient.length < 4) break;
            index++;
        }
    }

    
    return answer;
}
