const [n, k] = input[0].split(" ").map(Number);

class Queue {
  constructor() {
    this.items = {};
    this.headIndex = 0;
    this.tailIndex = 0;
  }
  enqueue(item) {
    //ë„£ê¸°
    this.items[this.tailIndex] = item;

  dequeue() {

  }
  getLength() {
    return this.tailIndex - this.headIndex;
  }
}
const MAX = 100001;
const visited = Array.from({ length: MAX }).fill(0);

function bfs() {
  const queue = new Queue();
  queue.enqueue(n);
  while (queue.getLength() != 0) {

  }
}
